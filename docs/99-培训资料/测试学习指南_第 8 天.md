**步骤 3: 应用初始化**
```python
# app.py
from flask_babel import Babel

babel = Babel()

def create_app():
    # ...
    def get_locale():
        # 1. URL 参数
        lang = request.args.get('lang')
        if lang in app.config['BABEL_SUPPORTED_LOCALES']:
            return lang
        
        # 2. Session
        lang = session.get('language')
        if lang in app.config['BABEL_SUPPORTED_LOCALES']:
            return lang
        
        # 3. 请求头
        return request.accept_languages.best_match(
            app.config['BABEL_SUPPORTED_LOCALES']
        ) or 'zh_CN'
    
    babel.init_app(app, locale_selector=get_locale)
```

```python
class Feedback(db.Model):
```

```python
@feedback_bp.route('/api/feedback', methods=['POST'])
@login_required
def submit_feedback():
    """提交用户反馈"""
    data = request.get_json()
    
    # 验证
    content = data.get('content', '').strip()
    if not content:
        return jsonify({'error': '反馈内容不能为空'}), 400
    
    if len(content) > 5000:
        return jsonify({'error': '内容过长'}), 400
    
    # 创建反馈
    feedback = Feedback(
        user_id=current_user.id,
        feedback_type=data.get('type', 'other'),
        title=data.get('title', ''),
        content=content,
        priority=data.get('priority', 'medium')
    )
    
    db.session.add(feedback)
    db.session.commit()
    
    return jsonify({
        'success': True,
        'message': '反馈已提交',
        'feedback_id': feedback.id
    }), 201
```

```python
# routes/analytics.py
@analytics_bp.route('/api/analytics/track', methods=['POST'])
def track_event():
    """追踪用户事件"""
    data = request.get_json()
    
    if not data or 'event_name' not in data:
        return jsonify({'error': '缺少 event_name'}), 400
    
    # 解析 User-Agent
    user_agent = request.headers.get('User-Agent', '')
    device_info = parse_user_agent(user_agent)
    
    # 创建事件
    event = UserEvent(
        user_id=current_user.id if current_user.is_authenticated else None,
        event_name=data['event_name'],
        event_category=data.get('event_category', 'custom'),
        device_type=device_info.get('device_type'),
        browser=device_info.get('browser'),
        os=device_info.get('os'),
        properties=json.dumps(data.get('properties', {}))
    )
    
    db.session.add(event)
    db.session.commit()
    
    return jsonify({'success': True, 'event_id': event.id}), 201
```

**批量提交优化**:

```python
@analytics_bp.route('/api/analytics/batch', methods=['POST'])
def batch_track_events():
    """批量追踪事件"""
    data = request.get_json()
    events_data = data['events']
    
    if len(events_data) > 100:
        return jsonify({'error': '单次最多100个事件'}), 400
    
    events = []
    for event_data in events_data:
        event = UserEvent(...)
        events.append(event)
    
    # 批量插入（性能优化）
    db.session.bulk_save_objects(events)
    db.session.commit()
    
    return jsonify({'count': len(events)}), 201
```

**统计分析**:
```python
@analytics_bp.route('/api/analytics/stats', methods=['GET'])
def get_analytics_stats():
    """获取统计数据"""
    days = request.args.get('days', 7, type=int)
    end_date = datetime.utcnow()
    start_date = end_date - timedelta(days=days)
    
    # 总事件数
    total_events = UserEvent.query.filter(
        UserEvent.created_at >= start_date
    ).count()
    
    # 按分类统计
    category_stats = db.session.query(
        UserEvent.event_category,
        func.count(UserEvent.id)
    ).filter(
        UserEvent.created_at >= start_date
    ).group_by(UserEvent.event_category).all()
    
    # TOP 20 热门事件
    top_events = db.session.query(
        UserEvent.event_name,
        func.count(UserEvent.id).label('count')
    ).group_by(UserEvent.event_name).order_by(
        func.count(UserEvent.id).desc()
    ).limit(20).all()
    
    return jsonify({
        'total_events': total_events,
        'by_category': dict(category_stats),
        'top_events': [{'event': e, 'count': c} for e, c in top_events]
    })
```

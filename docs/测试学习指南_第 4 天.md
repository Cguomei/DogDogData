### 2.3 将测试用例转化为代码

**模板结构：**

```python
"""
用户注册功能测试
"""
from .test_framework import test_case, test_manager, TestResult
from models import User


class TestUserRegistration:
    """用户注册功能测试类"""

    @test_case('TC-REG-001', priority='High', expected='注册成功')
    def test_register_success(self, client, db):
        """正常注册流程"""
        result = TestResult('TC-REG-001', 'test_register_success', '用户注册', 'High')
        result.expected_result = '注册成功，创建新用户并跳转到登录页'

        try:
            # 1. 执行注册操作
            response = client.post('/register', data={
                'username': 'testuser001',
                'password': 'TestPass123'
            }, follow_redirects=True)

            # 2. 验证响应
            assert response.status_code == 200

            # 3. 验证跳转（页面包含"登录"二字）
            response_text = response.get_data(as_text=True)
            assert '登录' in response_text

            # 4. 验证数据库中创建了用户
            user = User.query.filter_by(username='testuser001').first()
            assert user is not None
            assert user.username == 'testuser001'

            # 5. 记录成功
            result.status = 'PASS'
            result.actual_result = f'注册成功，用户 ID: {user.id}'

        except AssertionError as e:
            result.status = 'FAIL'
            result.error_message = str(e)
            result.actual_result = f'注册失败：{str(e)}'
            raise
        finally:
            test_manager.record_result(result)
```

## 三、测试数据的准备方法

### 3.1 使用 db fixture 创建数据

**方法 1：直接在测试中创建**

```python
def test_something(client, db):
    # 创建测试用户
    user = User(username='testuser')
    user.set_password('password123')
    db.session.add(user)
    db.session.commit()

    # 现在可以使用这个用户了
    response = client.post('/login', data={
        'username': 'testuser',
        'password': 'password123'
    })
```

**优点：**

- ✅ 简单直接
- ✅ 数据就在当前测试中
- ✅ 容易理解

**缺点：**

- ❌ 每个测试都要写一遍
- ❌ 代码重复

---

**方法 2：使用专门的 fixture**

在 conftest.py 中创建：

```python
@pytest.fixture
def test_user(db):
    """创建测试用户的 fixture"""
    user = User(username='fixture_user')
    user.set_password('password123')
    db.session.add(user)
    db.session.commit()

    return user  # 返回给用户


# 在测试中使用
def test_login(client, test_user):
    # test_user 会自动创建
    response = client.post('/login', data={
        'username': test_user.username,
        'password': 'password123'
    })
```

**优点：**

- ✅ 代码复用
- ✅ 一次定义，多处使用
- ✅ 自动清理

**缺点：**

- ❌ 需要额外定义 fixture
- ❌ 不够灵活

---

**方法 3：使用工厂模式（推荐）**

创建工具文件 `Test/test_tools/factories.py`：

```python
from models import User

class UserFactory:
    """用户工厂类"""

    @staticmethod
    def create(username=None, password=None, **kwargs):
        """创建用户的便捷方法"""
        if username is None:
            import uuid
            username = f'user_{uuid.uuid4().hex[:8]}'  # 生成随机用户名

        if password is None:
            password = 'default_password123'

        user = User(username=username, **kwargs)
        user.set_password(password)

        return user


class BreedFactory:
    """品种工厂类"""

    @staticmethod
    def create(name=None, **kwargs):
        """创建品种的"""
        if name is None:
            import uuid
            name = f'Breed_{uuid.uuid4().hex[:6]}'

        from models import Breed
        breed = Breed(name=name, **kwargs)

        return breed
```

**在测试中使用：**

```python
from test_tools.factories import UserFactory

def test_something(client, db):
    # 快速创建用户
    user = UserFactory.create(username='my_test_user')
    db.session.add(user)
    db.session.commit()

    # 或者用随机用户名
    user2 = UserFactory.create()  # 自动生成类似 user_a1b2c3d4 的用户名
    db.session.add(user2)
    db.session.commit()
```

**优点：**

- ✅ 超级灵活
- ✅ 可以覆盖默认值
- ✅ 支持随机数据
- ✅ 专业做法

**缺点：**

- ❌ 需要额外文件
- ❌ 需要学习成本

```python
from test_tools.factories import UserFactory
```

## 四、断言（Assert）完全指南

### 4.1 什么是断言？

**通俗解释：**
断言 = **检查点**

就像老师检查作业一样，断言用来检查测试结果是否符合预期。

### 4.2 常用断言类型

#### 类型 1：状态码断言

```python
# 检查 HTTP 状态码
assert response.status_code == 200      # 成功
assert response.status_code == 201      # 创建成功
assert response.status_code == 302      # 重定向
assert response.status_code == 400      # 请求错误
assert response.status_code == 401      # 未授权
assert response.status_code == 403      # 禁止访问
assert response.status_code == 404      # 未找到
assert response.status_code == 500      # 服务器错误
```

**实际例子：**

```python
def test_create_user(client):
    # 创建用户应该返回 201
    response = client.post('/users', data={'name': 'test'})
    assert response.status_code == 201

def test_delete_user(client):
    # 删除不存在的用户应该返回 404
    response = client.delete('/users/999')
    assert response.status_code == 404
```

---

#### 类型 2：内容断言

```python
# 检查响应中包含某个字符串
response_text = response.get_data(as_text=True)
assert '登录成功' in response_text
assert '错误' not in response_text

# 检查 JSON 响应
json_data = response.get_json()
assert json_data['success'] == True
assert json_data['user']['name'] == 'test'
assert 'id' in json_data['user']

# 检查 HTML 元素（高级用法）
from bs4 import BeautifulSoup
soup = BeautifulSoup(response_text, 'html.parser')
assert soup.find('h1').text == '首页'
```

**实际例子：**

```python
def test_login_success(client):
    response = client.post('/login', data={
        'username': 'test',
        'password': '123456'
    })

    response_text = response.get_data(as_text=True)

    # 检查是否包含成功关键词
    assert '登录成功' in response_text or '首页' in response_text
```

---

#### 类型 3：数据库断言

```python
# 检查记录是否存在
user = User.query.filter_by(username='test').first()
assert user is not None
assert user.email == 'test@example.com'

# 检查记录数量
count = User.query.count()
assert count == 5

# 检查关系
posts = user.posts.all()
assert len(posts) == 3
assert posts[0].title == '第一篇文章'
```

**实际例子：**

```python
def test_register_creates_user(client, db):
    response = client.post('/register', data={
        'username': 'newuser',
        'password': 'password123'
    })

    # 验证数据库中创建了用户
    user = User.query.filter_by(username='newuser').first()
    assert user is not None, "用户应该被创建"
    assert user.username == 'newuser'
```

---

#### 类型 4：异常断言

```python
# 检查是否抛出异常
import pytest

with pytest.raises(ValueError):
    int('not a number')

# 检查特定错误信息
with pytest.raises(AssertionError, match='期望的值'):
    assert False, "期望的值"
```

**实际例子：**

```python
def test_divide_by_zero():
    with pytest.raises(ZeroDivisionError):
        1 / 0
```

---

#### 类型 5：布尔值断言

```python
# 检查真假
assert user.is_active == True
assert user.is_admin == False

# 检查空值
assert result_list == []      # 空列表
assert result_dict == {}      # 空字典
assert text == ''             # 空字符串
assert value is None          # None 值

# 检查非空
assert user_list != []        # 非空列表
assert text != ''             # 非空字符串
```

---

#### 类型 6：数值范围断言

```python
# 检查范围
assert 0 <= percentage <= 100
assert age > 0
assert price >= 0

# 检查接近
assert abs(calculated - expected) < 0.01  # 误差小于 0.01
```

**实际例子：**

```python
def test_discount_calculation():
    original_price = 100
    discount = 0.2
    final_price = original_price * (1 - discount)

    assert 75 <= final_price <= 85  # 应该在合理范围内
```

---

### 4.3 断言的最佳实践

#### ✅ 好的断言

```python
# 1. 清晰明确
assert response.status_code == 200, "HTTP 状态码应该是 200"

# 2. 包含有用信息
assert user.email == 'test@example.com', f"邮箱应该是 test@example.com，实际是{user.email}"

# 3. 一次检查一件事
assert user.name == 'John'
assert user.age == 25
# 而不是
assert user.name == 'John' and user.age == 25  # 失败了不知道哪个错了

# 4. 使用有意义的消息
assert cart.total > 0, "购物车总金额应该大于 0"
```

#### ❌ 糟糕的断言

```python
# 1. 没有错误信息
assert x == 5  # 失败了不知道为什么

# 2. 检查太多东西
assert user.name == 'John' and user.age == 25 and user.email == 'john@example.com'

# 3. 永远为真的断言
assert True  # 毫无意义

# 4. 复杂的表达式
assert len([x for x in users if x.active]) > len([x for x in users if not x.active])
# 太难理解了
```

## 五、边界情况处理

### 5.1 什么是边界情况？

**通俗解释：**
边界情况 = **极端情况**

就像考试，不仅要看中等难度的题，还要看最简单和最难的题。

### 5.2 常见的边界情况

#### 边界 1：空值

```python
# 测试空字符串
def test_empty_username(self, client, db):
    """用户名为空"""
    response = client.post('/register', data={
        'username': '',  # 空字符串
        'password': 'TestPass123'
    })
    assert response.status_code == 400  # 应该拒绝
```

#### 边界 2：最大值

```python
# 测试最大长度
def test_max_length_username(self, client, db):
    """用户名达到最大长度"""
    max_username = 'a' * 20  # 20 个字符
    response = client.post('/register', data={
        'username': max_username,
        'password': 'TestPass123'
    })
    assert response.status_code == 200  # 应该成功
```

#### 边界 3：超过最大值

```python
# 测试超过最大长度
def test_too_long_username(self, client, db):
    """用户名超过最大长度"""
    too_long_username = 'a' * 21  # 21 个字符
    response = client.post('/register', data={
        'username': too_long_username,
        'password': 'TestPass123'
    })
    assert response.status_code == 400  # 应该拒绝
```

#### 边界 4：最小值

```python
# 测试最小长度
def test_min_length_username(self, client, db):
    """用户名刚好达到最小长度"""
    min_username = 'abcd'  # 4 个字符
    response = client.post('/register', data={
        'username': min_username,
        'password': 'TestPass123'
    })
    assert response.status_code == 200  # 应该成功
```

#### 边界 5：低于最小值

```python
# 测试低于最小长度
def test_too_short_username(self, client, db):
    """用户名小于最小长度"""
    too_short_username = 'abc'  # 3 个字符
    response = client.post('/register', data={
        'username': too_short_username,
        'password': 'TestPass123'
    })
    assert response.status_code == 400  # 应该拒绝
```

#### 边界 6：特殊字符

```python
# 测试 SQL 注入
def test_sql_injection_username(self, client, db):
    """SQL 注入攻击"""
    malicious_username = "' OR '1'='1"
    response = client.post('/register', data={
        'username': malicious_username,
        'password': 'TestPass123'
    })
    # 应该安全处理
    assert response.status_code in [200, 400]
```

#### 边界 7：Unicode 字符

```python
# 测试中文用户名
def test_chinese_username(self, client, db):
    """中文用户名"""
    chinese_username = '张三'
    response = client.post('/register', data={
        'username': chinese_username,
        'password': 'TestPass123'
    })
    # 根据需求决定应该接受还是拒绝
    assert response.status_code in [200, 400]
```

### 5.3 边界情况检查清单

**设计测试时问自己：**

## 六、完整的测试案例

### 6.1 案例：测试"添加商品到购物车"

**需求：**

```
用户可以将商品添加到购物车
- 商品 ID 必须有效
- 数量必须是正整数
- 库存不足时不能添加
- 同一商品可以多次添加（数量累加）
```

**完整测试代码：**

```python
"""
购物车功能测试
"""
from .test_framework import test_case, test_manager, TestResult
from models import Product, CartItem
from test_tools.factories import UserFactory


class TestShoppingCartFunctionality:
    """购物车功能测试类"""

    @test_case('TC-CART-001', priority='High', expected='成功添加商品')
    def test_add_product_to_cart_success(self, client, db, logged_in_client):
        """正常添加商品到购物车"""
        result = TestResult('TC-CART-001', 'test_add_product_to_cart_success', '购物车', 'High')
        result.expected_result = '商品成功添加到购物车'

        try:
            # 1. 准备测试数据
            product = Product(
                name='测试商品',
                price=99.99,
                stock=100  # 库存充足
            )
            db.session.add(product)
```

**错误 2：断言太复杂**

```python
# ❌ 不好
assert user.name == 'John' and user.age == 25 and user.email == 'john@example.com'

# ✅ 好
assert user.name == 'John'
assert user.age == 25
assert user.email == 'john@example.com'
```

```python
# ❌ 只检查了状态码

assert '成功' in response.text
```

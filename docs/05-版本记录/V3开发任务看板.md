```python
# routes/ai_assistant.py
def get_session_context(session_id, max_messages=10):
    """获取会话上下文"""
    history = ChatMessage.query.filter_by(
        session_id=session_id
    ).order_by(
        ChatMessage.created_at.desc()
    ).limit(max_messages).all()
    
    history.reverse()
    
    return [
        {"role": msg.role, "content": msg.content}
        for msg in history
    ]

# 在chat函数中调用
context = get_session_context(session_id)
full_messages = context + [{"role": "user", "content": user_message}]
answer = call_local_model(full_messages)
```

```python
# utils/knowledge_base.py
def auto_learn(self, question, answer, question_type, confidence=0.9):
    """从模型回答中自动学习"""
    if confidence > 0.8:
        # 检查是否已存在相似问题
        similar = self.search_similar(question, threshold=0.85)
        
        if not similar:
            self.add_knowledge(
                question=question,
                answer=answer,
                category=question_type,
                source='model',
                confidence=confidence
            )
            logger.info(f"✅ 自动学习: {question[:50]}...")
            return True
    
    return False
```

```python
# utils/alert_system.py
from apscheduler.schedulers.background import BackgroundScheduler

def check_price_alerts():
    """检查价格异常"""
    breeds = DogBreed.query.all()
    
    for breed in breeds:
        current_avg = get_current_avg_price(breed.name)
        last_week_avg = get_last_week_avg_price(breed.name)
        
        change_rate = (current_avg - last_week_avg) / last_week_avg
        
        if change_rate > 0.15:
            send_alert(
                type="price_increase",
                breed=breed.name,
                change_rate=change_rate
            )

scheduler = BackgroundScheduler()
scheduler.add_job(check_price_alerts, 'cron', hour=2)
scheduler.start()
```

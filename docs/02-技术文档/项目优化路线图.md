```python
@test_case('TC-API-EDGE-002', priority='High')
def test_concurrent_requests():
    """测试并发请求处理"""
    import concurrent.futures

    def add_breed(index):
        return api_client.post('/api/breeds', json={
            'breed_name': f'Breed_{index}',
            'size_category': '中型'
        })

    # 执行: 50 个并发请求
    with concurrent.futures.ThreadPoolExecutor(max_workers=10) as executor:
        futures = [executor.submit(add_breed, i) for i in range(50)]
        responses = [f.result() for f in futures]

    # 验证: 至少有一个成功，无服务器错误
    success_count = sum(1 for r in responses if r.status_code == 201)
    error_500_count = sum(1 for r in responses if r.status_code == 500)

    assert success_count > 0
    assert error_500_count == 0
```

```python
@pytest.mark.performance
def test_api_response_time_under_load():
    """测试 API 在负载下的响应时间"""
    import requests
    import statistics

    response_times = []

    # 发送 100 个请求
    for i in range(100):
        start = time.time()
        response = requests.get('http://localhost:5000/api/breeds')
        elapsed = (time.time() - start) * 1000  # 转换为毫秒
        response_times.append(elapsed)

    # 统计指标
    avg_time = statistics.mean(response_times)
    p95_time = statistics.quantiles(response_times, n=20)[18]  # 95 百分位
    max_time = max(response_times)

    print(f"\n性能统计:")
    print(f"  平均响应时间: {avg_time:.2f}ms")
    print(f"  P95 响应时间: {p95_time:.2f}ms")
    print(f"  最大响应时间: {max_time:.2f}ms")

    # 断言：P95 应小于 500ms
    assert p95_time < 500, f"P95 响应时间超标: {p95_time}ms"
```

```python
# utils/timezone.py
from datetime import datetime
import pytz

def get_user_timezone():
    """获取用户时区（从 Session 或 Cookie）"""
    from flask import session
    return session.get('timezone', 'Asia/Shanghai')

def format_datetime(dt, format='medium'):
    """格式化时间为用户时区"""
    user_tz = pytz.timezone(get_user_timezone())
    localized_dt = dt.astimezone(user_tz)

    if format == 'medium':
        return localized_dt.strftime('%Y-%m-%d %H:%M:%S')
    elif format == 'short':
        return localized_dt.strftime('%m-%d %H:%M')
```

```html
<!-- 原 jQuery 实现 → Alpine.js -->
<div x-data="petComponent()" 
     x-init="init()"
     class="pet-container">

    <!-- 宠物形象 -->
    <div class="pet" 
         :class="petState"
         @click="interact('touch')">
        <span x-text="petEmoji"></span>
    </div>

    <!-- 控制按钮 -->
    <div class="pet-controls">
        <button @click="interact('feed')">🍖 {{ _('喂食') }}</button>
        <button @click="interact('play')">🎾 {{ _('玩耍') }}</button>
        <button @click="interact('sleep')">😴 {{ _('休息') }}</button>
    </div>

    <!-- 状态提示 -->
    <div x-show="showMessage" 
         x-transition
         x-text="message"
         class="pet-message">
    </div>
</div>

<script>
function petComponent() {
    return {
        petState: 'idle',
        petEmoji: '🐶',
        message: '',
        showMessage: false,

        init() {
            // 从 localStorage 加载状态
            const saved = localStorage.getItem('pet_state');
            if (saved) {
                Object.assign(this, JSON.parse(saved));
            }

            // 定时保存状态
            setInterval(() => this.saveState(), 5000);
        },

        interact(action) {
            switch(action) {
                case 'feed':
                    this.petState = 'eating';
                    this.petEmoji = '😋';
                    this.showMessage = true;
                    this.message = '{{ _("好吃！") }}';
                    break;
                case 'play':
                    this.petState = 'playing';
                    this.petEmoji = '🎉';
                    break;
                case 'sleep':
                    this.petState = 'sleeping';
                    this.petEmoji = '💤';
                    break;
            }

            // 2 秒后恢复
            setTimeout(() => {
                this.petState = 'idle';
                this.petEmoji = '🐶';
                this.showMessage = false;
            }, 2000);
        },

        saveState() {
            localStorage.setItem('pet_state', JSON.stringify({
                petState: this.petState,
                petEmoji: this.petEmoji
            }));
        }
    }
}
</script>
```

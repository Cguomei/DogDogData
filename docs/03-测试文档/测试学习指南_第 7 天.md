```python
    # ========== 功能测试 ==========
    
    @test_case('TC-ORDER-001', priority='Critical', expected='订单创建成功')
    def test_create_order_success(self, client, db, logged_in_client):
        """正常下单流程"""
        result = TestResult('TC-ORDER-001', 'test_create_order_success', '订单系统', 'Critical')
        result.expected_result = '订单创建成功，库存正确扣减'

        try:

            product = Product(name='iPhone 15', price=5999.00, stock=10)
            db.session.add(product)
            db.session.commit()
            
            product_id = product.id

            # 2. 执行下单操作
            order_data = {
                'product_id': product_id,
                'quantity': 2,
                'address': '北京市朝阳区 xxx'
            }

            response = logged_in_client.post('/api/orders/create',
                                             json=order_data,

            # 3. 验证响应
            assert response.status_code == 201
            json_data = response.get_json()
            assert 'order_id' in json_data['data']
            assert json_data['data']['status'] == 'pending'

            result.actual_result = f'订单创建成功，ID: {order.id}'

    # ========== 性能测试 ==========
    
    @test_case('TC-ORDER-006', priority='High', expected='并发下单正确处理')
    def test_concurrent_order_placement(self, client, db, logged_in_client):
        """并发下单测试（最后一件商品）"""
        result = TestResult('TC-ORDER-006', 'test_concurrent_order_placement', '订单系统', 'High')
        result.expected_result = '只有 1 个订单成功，库存正确扣减'

        try:
            import threading
            
            # 1. 创建只有 1 件库存的商品
            product = Product(name='抢购商品', price=1.00, stock=1)
            db.session.add(product)
            db.session.commit()
            
            product_id = product.id
            success_count = 0
            results = []

            # 2. 定义下单函数
            def place_order():
                nonlocal success_count
                order_data = {
                    'product_id': product_id,
                    'quantity': 1
                }
                
                try:
                    response = logged_in_client.post('/api/orders/create',
                                                     json=order_data,
                                                     content_type='application/json')
                    if response.status_code == 201:
                        success_count += 1
                        results.append('success')
                    else:
                        results.append('failed')
                except Exception as e:
                    results.append(str(e))

            # 3. 创建 5 个线程同时下单
            threads = []
            for i in range(5):
                t = threading.Thread(target=place_order)
                threads.append(t)

            # 4. 同时启动所有线程
            for t in threads:
                t.start()

            # 5. 等待所有线程完成
            for t in threads:
                t.join()

            # 6. 验证：只有 1 个订单成功
            assert success_count == 1, \
                f"期望 1 个成功，实际{success_count}个成功。结果：{results}"

            # 7. 验证库存已扣减为 0
            db.session.refresh(product)
            assert product.stock == 0

            result.status = 'PASS'
            result.actual_result = f'并发控制正确：{success_count}个成功，{5-success_count}个失败'

        except AssertionError as e:
            result.status = 'FAIL'
            result.error_message = str(e)
            result.actual_result = f'并发控制失效：{success_count}个成功'
            raise
        finally:
            test_manager.record_result(result)
```

```python
"""
测试报告生成器 - 增强版
"""
from datetime import datetime
import json


class AdvancedReportGenerator:
    """高级报告生成器"""

    def __init__(self, test_manager):
        self.test_manager = test_manager

    def generate_html_report(self, filename):
        """生成详细的 HTML 测试报告"""
        summary = self.test_manager.get_summary()
        results = self.test_manager.results

        modules = {}

            module = result.module_name
            if module not in modules:
                modules[module] = {'total': 0, 'passed': 0, 'failed': 0}
            
            modules[module]['total'] += 1
            if result.status == 'PASS':
                modules[module]['passed'] += 1
            else:
                modules[module]['failed'] += 1

        html = f'''
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>测试报告 - {datetime.now().strftime("%Y-%m-%d %H:%M")}</title>

<body>
    <div class="header">
        <h1>📊 自动化测试报告</h1>
        <p>生成时间：{datetime.now().strftime("%Y-%m-%d %H:%M:%S")}</p>

    def generate_json_report(self, filename):
        """生成 JSON 格式报告"""
        summary = self.test_manager.get_summary()
        
        data = {
            'report_info': {
                'generated_at': datetime.now().isoformat(),
                'tester': self.test_manager.tester_name,
                'total_duration': summary['execution_time']
            },
            'summary': summary,
            'test_cases': [r.to_dict() for r in self.test_manager.results]
        }

        with open(filename, 'w', encoding='utf-8') as f:
            json.dump(data, f, ensure_ascii=False, indent=2)

        print(f'✓ JSON 报告已生成：{filename}')

    def analyze_and_recommend(self):
        """分析测试结果并给出改进建议"""
        results = self.test_manager.results
        failed_cases = [r for r in results if r.status == 'FAIL']
        
        print('\n' + '='*60)
        print('📊 测试分析报告')
        print('='*60)
        
        # 1. 总体情况
        total = len(results)
        passed = sum(1 for r in results if r.status == 'PASS')
        pass_rate = (passed / total * 100) if total > 0 else 0
        
        print(f'\n✅ 总体表现：{"优秀" if pass_rate >= 95 else "良好" if pass_rate >= 80 else "需改进"}')
        print(f'   通过率：{pass_rate:.1f}%')
        
        # 2. 失败分析
        if failed_cases:
            print(f'\n❌ 发现 {len(failed_cases)} 个失败用例：')
            
            # 按优先级排序
            priority_order = {'Critical': 0, 'High': 1, 'Medium': 2, 'Low': 3}
            failed_cases.sort(key=lambda x: priority_order.get(x.priority, 4))
            
            for case in failed_cases[:5]:  # 只显示前 5 个
                print(f'   - [{case.priority}] {case.case_id}: {case.method_name}')
                print(f'     错误：{case.error_message[:100]}')
        
        # 3. 性能分析
        slow_tests = [r for r in results if r.execution_time > 1.0]
        if slow_tests:
            print(f'\n⚠️  发现 {len(slow_tests)} 个慢测试（>1 秒）：')
            for test in slow_tests[:3]:
                print(f'   - {test.method_name}: {test.execution_time:.2f}秒')
        
        # 4. 改进建议
        print('\n💡 改进建议：')
        
        if pass_rate < 95:
            print('   1. 优先修复 Critical 和 High 级别的失败用例')
        
        if len(slow_tests) > 0:
            print('   2. 优化慢测试的执行效率')
        
        critical_failures = [c for c in failed_cases if c.priority == 'Critical']
        if critical_failures:
            print('   3. ⚠️  存在 Critical 级别失败，建议立即修复！')
        
        if pass_rate >= 95 and len(slow_tests) == 0:
            print('   🎉 测试质量很高，继续保持！')
        
        print('='*60 + '\n')


# 使用示例

generator.generate_html_report('reports/final_test_report.html')
generator.generate_json_report('reports/final_test_report.json')
```
